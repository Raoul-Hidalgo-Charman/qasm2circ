% Sample qasm input file - EPR creation
%
%         qubit 	q0;\phi;
%         qubit 	q1;;\psi
% 
% 	h	q0	% create EPR pair
% 	cnot	q0,q1

%  Time 01:
%    Gate 00 h(q0)
%  Time 02:
%    Gate 01 cnot(q0,q1)

% Qubit circuit matrix:
%
% q0: gAxA, gBxA, n  
% q1: n  , gBxB, n  

\documentclass[11pt]{article}
\input{xyqcirc.tex}

% definitions for the circuit elements

\def\gAxA{\op{H}\w\A{gAxA}}
\def\gBxA{\b\w\A{gBxA}}
\def\gBxB{\o\w\A{gBxB}}

% definitions for bit labels and initial states

\def\bA{\qrule \q{\phi}}
\def\bB{\qrule }
\def\fA{\qrule}
\def\fB{\qrule\q{\psi}}

% The quantum circuit as an xymatrix

\xymatrix@R=5pt@C=10pt{
    \bA & \gAxA &\gBxA &\n   & \fA
\\  \bB & \n   &\gBxB &\n   & \fB
%
% vertical lines and other post-xymatrix latex
%
\ar@{-}"gBxB";"gBxA"
}

\end{document}
