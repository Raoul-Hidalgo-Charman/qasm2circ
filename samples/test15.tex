\documentclass[11pt]{article}

% definitions for the circuit elements

\def\gAxA{*-{}\w\A{gAxA}}
\def\gBxA{\b\w\A{gBxA}}
\def\gBxB{\b\w\A{gBxB}}
\def\gCxA{*-{}\w\A{gCxA}}
\def\gDxA{\dmeterwide{HZ_\theta}{18pt}\w\A{gDxA}}

% definitions for bit labels and initial states

\def\bA{\qrule \q{\psi}}
\def\bB{\qrule \q{+}}
\def\fA{\qrule}
\def\fB{\qrule}

% The quantum circuit as an xymatrix

\xymatrix@R=5pt@C=10pt{
    \bA & \gAxA &\gBxA &\gCxA &\gDxA &\l   & \fA
\\  \bB & \n   &\gBxB &\n   &\n   &\n   & \fB
%
% vertical lines and other post-xymatrix latex

\nv{gBxA}{gBxB}
