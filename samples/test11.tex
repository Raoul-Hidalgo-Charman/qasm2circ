% Qubit circuit matrix:
%
% q0: gAxA, gBxA, n  , gDxA, n  
% q1: n  , gBxB, gCxB, gDxB, n  
% q2: n  , n  , n  , gDxC, n  

\documentclass[11pt]{article}
\input{xyqcirc.tex}

% definitions for the circuit elements

\def\gAxA{\op{H}\w\A{gAxA}}
\def\gBxA{\gnqubit{U_{f(x)}}{d}\w\A{gBxA}}
\def\gBxB{\gspace{U_{f(x)}}\w\A{gBxB}}
\def\gCxB{\op{H}\w\A{gCxB}}
\def\gDxA{\gnqubit{U_{f(x,y)}}{dd}\w\A{gDxA}}
\def\gDxB{\gspace{U_{f(x,y)}}\w\A{gDxB}}
\def\gDxC{\gspace{U_{f(x,y)}}\w\A{gDxC}}

% definitions for bit labels and initial states

\def\bA{\qrule }
\def\bB{\qrule }
\def\bC{\qrule }
\def\fA{\qrule}
\def\fB{\qrule}
\def\fC{\qrule}

% The quantum circuit as an xymatrix

\xymatrix@R=5pt@C=10pt{
    \bA & \gAxA &\gBxA &\n   &\gDxA &\n   & \fA
\\  \bB & \n   &\gBxB &\gCxB &\gDxB &\n   & \fB
\\  \bC & \n   &\n   &\n   &\gDxC &\n   & \fC
%
% vertical lines and other post-xymatrix latex
%
}
