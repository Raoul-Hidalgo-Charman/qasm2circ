% Sample qasm input file - EPR creation
%
% 
%     mcbit   c0;c0
%     mqubit 	q0;q0
%     qubit 	q1;q1
% 
%     h	q0	% create EPR pair
%     cnot	q0,q1
%     toffoli c0,q0,q1

%  Time 01:
%    Gate 00 h(q0)
%  Time 02:
%    Gate 01 cnot(q0,q1)
%  Time 03:
%    Gate 02 toffoli(c0,q0,q1)

% Qubit circuit matrix:
%
% c0: L  , L  , gCxA, L  
% q0: gAxB, gBxB, gCxB, N  
% q1: n  , gBxC, gCxC, n  

\documentclass[11pt]{article}
\input{xyqcirc.tex}

% definitions for the circuit elements

\def\gAxB{\op{H}\W\A{gAxB}}
\def\gBxB{\b\W\A{gBxB}}
\def\gBxC{\o\w\A{gBxC}}
\def\gCxA{\b\C\A{gCxA}}
\def\gCxB{\b\W\A{gCxB}}
\def\gCxC{\o\w\A{gCxC}}

% definitions for bit labels and initial states

\def\bA{\qrule c_{0}}
\def\bB{\qrule \q{q_{0}}}
\def\bC{\qrule \q{q_{1}}}
\def\fA{\qrule}
\def\fB{\qrule}
\def\fC{\qrule}

% The quantum circuit as an xymatrix

\xymatrix@R=5pt@C=10pt{
    \bA & \L   &\L   &\gCxA &\L   & \fA
\\  \bB & \gAxB &\gBxB &\gCxB &\N   & \fB
\\  \bC & \n   &\gBxC &\gCxC &\n   & \fC
%
% vertical lines and other post-xymatrix latex
%
\nv{gBxC}{gBxB}
\lv{gCxC}{gCxA}\lv{gCxC}{gCxB}
}

\end{document}
